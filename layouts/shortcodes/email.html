{{- /* Spam-safe email link. The address is base64-encoded into a single opaque data-e attribute and assembled client-side (see partials/custom/head-end.html), so the raw HTML never contains a harvestable address or a mailto: href. The visible fallback uses " [at] " so it does not match an email regex either. Usage: {{< email "info@collectivedynamics.org" >}} or {{< email "addr" "label" >}} */ -}} {{- $addr := .Get 0 -}} {{- $parts := split $addr "@" -}} {{- $u := index $parts 0 -}} {{- $d := index $parts 1 -}} {{- $label := .Get 1 -}} {{- $enc := base64Encode $addr -}} {{- if $label -}} {{ $label }} {{- else -}} {{ $u }} [at] {{ $d }} {{- end -}}
